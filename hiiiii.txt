local function makeDraggable(frame)
	local dragging = false
	local dragInput, dragStart, startPos

	frame.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = frame.Position

			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)

	frame.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if dragging and input == dragInput then
			local delta = input.Position - dragStart
			frame.Position = UDim2.new(
				startPos.X.Scale,
				startPos.X.Offset + delta.X,
				startPos.Y.Scale,
				startPos.Y.Offset + delta.Y
			)
		end
	end)
end

local clicking = false

local function clickLoop()
	while clicking do
		-- ç¢ºä¿ UI æ›´æ–°å®Œæˆå†æŠ“ä½ç½®
		task.wait()

		local x = icon.AbsolutePosition.X + icon.AbsoluteSize.X / 2
		local y = icon.AbsolutePosition.Y + icon.AbsoluteSize.Y / 2

		VirtualInputManager:SendMouseButtonEvent(x, y, 0, true, game, 0)
		VirtualInputManager:SendMouseButtonEvent(x, y, 0, false, game, 0)

		local delayTime = tonumber(box.Text)
		if not delayTime or delayTime < 0 then delayTime = 0 end
		task.wait(delayTime)
	end
end

button.MouseButton1Click:Connect(function()
	clicking = not clicking
	if clicking then
		button.Text = "é—œé–‰"
		button.BackgroundColor3 = Color3.fromRGB(255, 80, 80)
		task.spawn(clickLoop)
	else
		button.Text = "é–‹å§‹"
		button.BackgroundColor3 = Color3.fromRGB(0, 200, 255)
	end
end)
